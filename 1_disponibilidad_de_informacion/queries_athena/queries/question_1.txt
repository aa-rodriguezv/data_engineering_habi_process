CREATE OR REPLACE VIEW QUERY_1_CITY_DISTRIBUTION AS 
SELECT COALESCE(CIUDAD, 'UNKNOWN') AS CIUDAD
    , COUNT(DISTINCT ID) AS CUENTA_BIENES
    , MIN(VALOR_ARRIENDO) AS MIN_ARRIENDO
    , MAX(VALOR_ARRIENDO) AS MAX_ARRIENDO
    , MIN(VALOR_VENTA) AS MIN_VENTA
    , MAX(VALOR_VENTA) AS MAX_VENTA
FROM "angel_test"."goods_data_cleaned_vw"
GROUP BY 1
ORDER BY 2 DESC
;
